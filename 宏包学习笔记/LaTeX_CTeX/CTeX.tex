\documentclass{article}

%rfr 可以参考ctex宏包的官方手册：https://sg.mirrors.cicku.me/ctan/language/chinese/ctex/ctex.pdf
\usepackage[scheme=plain]{ctex}
    % 设置"scheme=plain"（默认值为chinese），表示只提供中文支持功能，而不对文章版式进行任何修改（字号、行距、标题/章节样式等等）

\ctexset{space=true,% 
    % 设置"space = true"，表示在打印出的pdf中保留汉字之间的空格；
    % 设置"space = false"，且使用pdflatex编译时，表示在打印出的pdf中忽略汉字之间的空格，使用xelatex编译时，相当于auto模式；
    % 默认值"space = auto"模式下，打印出的pdf中会忽略汉字之间的空格
    %* 以上内容来自宏包管手册，可以发现，不管是使用pdflatex编译还是使用xelatex编译，似乎设置"space = false"和"space = auto"的效果相同
    %* 可以发现，汉字和字母之间不管代码中是否空格，打印效果中始终存在空格，可以通过插入空箱子"\mbox{}"来消除空格
today={old},% 
    % 设置"today=old"，保持日期的旧有格式，不进行汉化
}
